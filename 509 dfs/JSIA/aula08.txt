8. Considere a função calcularMediaNotas que utiliza o rest operator:
        function calcularMediaNotas(aluno, ...notas) {
            if (notas.length === 0) return 0;
            const soma = notas.reduce((acc, curr) => acc + curr, 0);
            return soma / notas.length;
        }
        console.log(calcularMediaNotas("Fernando", 7, 8, 9, 6));

        B. 8 (CORRETA)
